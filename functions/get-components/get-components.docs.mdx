import { GetComponents } from './get-components';

## Description

Return an object containing Selected components

### Function sage

```js
import { GetComponents } from '@hoomies/unagui.functions.get-components';

const UnaguiComponents = GetComponents({ names: ['UnaguiHero', 'UnaguiSection'] });
// result = { h1: [Function: h1], p: [Function: p], ...,'UnaguiHero': Function, 'UnaguiSection': Function}

const UnaguiComponentsOnly = GetComponents({ names: ['UnaguiHero', 'UnaguiSection'], withMDX: false });
// result = { 'UnaguiHero': Function, 'UnaguiSection': Function}
```
