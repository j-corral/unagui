import { GetComponents } from './get-components';

## Description

Return an object containing Selected components

### Function sage

```js
import { GetComponents } from '@hoomies/unagui.functions.get-components';

const UnaguiComponents = GetComponents({ names: ['Unagui.Hero', 'Unagui.Section'] });
// result = { h1: [Function: h1], p: [Function: p], ...,'Unagui.Hero': Function, 'Unagui.Section': Function}

const UnaguiComponentsOnly = GetComponents({ names: ['Unagui.Hero', 'Unagui.Section'], withMDX: false });
// result = { 'Unagui.Hero': Function, 'Unagui.Section': Function}
```
