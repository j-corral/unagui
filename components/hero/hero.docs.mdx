import { Hero } from './hero';
import { Wrapper } from '@hoomies/unagui.theme.wrapper';

### Component usage

```tsx
import { Hero } from '@hoomies/unagui.components.hero';

<Hero title="Awesome Title" />;
```

### Using props to customize the text

Modify the text to see it change live:

```js live
<Wrapper resetCSS>
  <Hero title="Live text" section={{ h: '250' }} />
</Wrapper>
```

### More props

The Hero component uses the [Section Component](https://bit.dev/hoomies/unagui/components/section) at root level.

The Hero component is extended from the [Chakra UI - Heading](https://chakra-ui.com/docs/typography/heading)
